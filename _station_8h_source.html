189  unsigned int minSlewTime = 0;
190  if ( dataWriteRate.is_initialized() ) {
191  double writeRate = *dataWriteRate;
192  double fraction = totalRecordingRate / writeRate - 1;
193  if ( fraction > 0 ) {
194  minSlewTime = static_cast<unsigned int>( ceil( static_cast<double>( observingTime ) * fraction ) );
195  }
