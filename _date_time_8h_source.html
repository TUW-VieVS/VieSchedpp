296  {

455  void FromTicks(int& year, int& month, int& day) const
456  {
457  int totalDays = static_cast<int>(m_encoded / TicksPerDay);
458 
459  /*
460  * number of 400 year cycles
461  */
462  int num400 = totalDays / 146097;
463  totalDays -= num400 * 146097;
464  /*
465  * number of 100 year cycles
466  */
467  int num100 = totalDays / 36524;
468  if (num100 == 4)
469  {
470  /*
471  * last day of the last leap century
472  */
473  num100 = 3;
474  }
475  totalDays -= num100 * 36524;
476  /*
477  * number of 4 year cycles
478  */
479  int num4 = totalDays / 1461;
480  totalDays -= num4 * 1461;
481  /*
482  * number of years
483  */
484  int num1 = totalDays / 365;
485  if (num1 == 4)
486  {
487  /*
488  * last day of the last leap olympiad
489  */
490  num1 = 3;
491  }
492  totalDays -= num1 * 365;
493 
494  /*
495  * find year
496  */
497  year = (num400 * 400) + (num100 * 100) + (num4 * 4) + num1 + 1;
498 
499  /*
500  * convert day of year to month/day
501  */
502  const int* daysInMonthPtr;
503  if (IsLeapYear(year))
504  {
505  daysInMonthPtr = daysInMonth[1];
506  }
507  else
508  {
509  daysInMonthPtr = daysInMonth[0];
510  }
511 
512  month = 1;
513  while (totalDays >= daysInMonthPtr[month] && month <= 12)
514  {
515  totalDays -= daysInMonthPtr[month++];
516  }
517 
518  day = totalDays + 1;
519  }
